# Python code for basic robot movement control
# Import the robot control library (simulated library for demonstration)
import robot_control_lib as rcl

# Initialize the robot
robot = rcl.Robot()

def basic_movement_demo():
    """Demo of robot's basic movement functions"""
    print("Robot starts moving forward...")
    robot.move_forward(duration=3)  # Move forward for 3 seconds
    
    print("Robot starts turning left...")
    robot.turn_left(duration=1)     # Turn left for 1 second
    
    print("Robot starts moving backward...")
    robot.move_backward(duration=2) # Move backward for 2 seconds
    
    print("Robot starts turning right...")
    robot.turn_right(duration=1)    # Turn right for 1 second
    
    print("Experiment completed, robot stops moving.")
    robot.stop()

# Call the demo function
if __name__ == "__main__":
    basic_movement_demo()
